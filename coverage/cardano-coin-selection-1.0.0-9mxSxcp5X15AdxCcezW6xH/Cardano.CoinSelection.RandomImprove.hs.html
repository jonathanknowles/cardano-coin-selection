   55 -- | An implementation of the __Random-Improve__ coin selection algorithm.
   56 --
   57 -- = Overview
   58 --
   59 -- The __Random-Improve__ coin selection algorithm works in __two phases__, by
   60 -- /first/ selecting UTxO entries /at random/ to pay for each of the given
   61 -- outputs, and /then/ attempting to /improve/ upon each of the selections.
   62 --
   63 -- === Phase 1: Random Selection
   64 --
   65 -- __In this phase, the algorithm randomly selects a minimal set of UTxO__
   66 -- __entries to pay for each of the given outputs.__
   67 --
   68 -- During this phase, the algorithm:
   69 --
   70 --   *  processes outputs in /descending order of coin value/.
   71 --
   72 --   *  maintains a /remaining UTxO set/, initially equal to the given
   73 --      /UTxO set/ parameter.
   74 --
   75 -- For each output of value __/v/__, the algorithm /randomly/ selects entries
   76 -- from the /remaining UTxO set/, until the total value of selected entries is
   77 -- greater than or equal to __/v/__. The selected entries are then associated
   78 -- with that output, and removed from the /remaining UTxO set/.
   79 --
   80 -- This phase ends when every output has been associated with a selection of
   81 -- UTxO entries.
   82 --
   83 -- However, if the remaining UTxO set is completely exhausted before all
   84 -- outputs can be processed, the algorithm terminates with an error.
   85 --
   86 -- === Phase 2: Improvement
   87 --
   88 -- __In this phase, the algorithm attempts to improve upon each of the UTxO__
   89 -- __selections made in the previous phase, by conservatively expanding the__
   90 -- __selection made for each output.__
   91 --
   92 -- During this phase, the algorithm:
   93 --
   94 --   *  processes outputs in /ascending order of coin value/.
   95 --
   96 --   *  continues to maintain the /remaining UTxO set/ produced by the previous
   97 --      phase.
   98 --
   99 --   *  maintains an /accumulated coin selection/, which is initially /empty/.
  100 --
  101 -- For each output of value __/v/__, the algorithm:
  102 --
  103 --  1.  __Calculates a /target range/__ for the total value of inputs used to
  104 --      pay for that output, defined by the triplet:
  105 --
  106 --      (/minimum/, /ideal/, /maximum/) = (/v/, /2v/, /3v/)
  107 --
  108 --  2.  __Attempts to /improve/ upon the /existing UTxO selection/__ for that
  109 --      output, by repeatedly selecting additional entries at random from the
  110 --      /remaining UTxO set/, stopping when the selection can be improved upon
  111 --      no further.
  112 --
  113 --      A selection with value /v1/ is considered to be an /improvement/ over a
  114 --      selection with value /v0/ if __all__ of the following conditions are
  115 --      satisfied:
  116 --
  117 --       * __Condition 1__: we have moved closer to the /ideal/ value:
  118 --
  119 --             abs (/ideal/ − /v1/) < abs (/ideal/ − /v0/)
  120 --
  121 --       * __Condition 2__: we have not exceeded the /maximum/ value:
  122 --
  123 --             /v1/ ≤ /maximum/
  124 --
  125 --       * __Condition 3__: when counting cumulatively across all outputs
  126 --       considered so far, we have not selected more than the /maximum/ number
  127 --       of UTxO entries specified by 'limit'.
  128 --
  129 --  3.  __Creates a /change value/__ for the output, equal to the total value
  130 --      of the /final UTxO selection/ for that output minus the value /v/ of
  131 --      that output.
  132 --
  133 --  4.  __Updates the /accumulated coin selection/__:
  134 --
  135 --       * Adds the /output/ to 'outputs'.
  136 --       * Adds the /improved UTxO selection/ to 'inputs'.
  137 --       * Adds the /change value/ to 'change'.
  138 --
  139 -- This phase ends when every output has been processed, __or__ when the
  140 -- /remaining UTxO set/ has been exhausted, whichever occurs sooner.
  141 --
  142 -- = Termination
  143 --
  144 -- When both phases are complete, the algorithm terminates.
  145 --
  146 -- The /accumulated coin selection/ and /remaining UTxO set/ are returned to
  147 -- the caller.
  148 --
  149 -- === Failure Modes
  150 --
  151 -- The algorithm terminates with an __error__ if:
  152 --
  153 --  1.  The /total value/ of the initial UTxO set (the amount of money
  154 --      /available/) is /less than/ the total value of the output list (the
  155 --      amount of money /required/).
  156 --
  157 --      See: __'InputValueInsufficientError'__.
  158 --
  159 --  2.  The /number/ of entries in the initial UTxO set is /smaller than/ the
  160 --      number of requested outputs.
  161 --
  162 --      Due to the nature of the algorithm, /at least one/ UTxO entry is
  163 --      required /for each/ output.
  164 --
  165 --      See: __'InputCountInsufficientError'__.
  166 --
  167 --  3.  Due to the particular /distribution/ of values within the initial UTxO
  168 --      set, the algorithm depletes all entries from the UTxO set /before/ it
  169 --      is able to pay for all requested outputs.
  170 --
  171 --      See: __'InputsExhaustedError'__.
  172 --
  173 --  4.  The /number/ of UTxO entries needed to pay for the requested outputs
  174 --      would /exceed/ the upper limit specified by 'limit'.
  175 --
  176 --      See: __'InputLimitExceededError'__.
  177 --
  178 -- = Motivating Principles
  179 --
  180 -- There are several motivating principles behind the design of the algorithm.
  181 --
  182 -- === Principle 1: Dust Management
  183 --
  184 -- The probability that random selection will choose dust entries from a UTxO
  185 -- set increases with the proportion of dust in the set.
  186 --
  187 -- Therefore, for a UTxO set with a large amount of dust, there's a high
  188 -- probability that a random subset will include a large amount of dust.
  189 --
  190 -- === Principle 2: Change Management
  191 --
  192 -- Ideally, coin selection algorithms should, over time, create a UTxO set that
  193 -- has /useful/ outputs: outputs that will allow us to process future payments
  194 -- with a minimum number of inputs.
  195 --
  196 -- If for each payment request of value __/v/__ we create a change output of
  197 -- /roughly/ the same value __/v/__, then we will end up with a distribution of
  198 -- change values that matches the typical value distribution of payment
  199 -- requests.
  200 --
  201 -- === Principle 3: Performance Management
  202 --
  203 -- Searching the UTxO set for additional entries to improve our change outputs
  204 -- is /only/ useful if the UTxO set contains entries that are sufficiently
  205 -- small enough. But it is precisely when the UTxO set contains many small
  206 -- entries that it is less likely for a randomly-chosen UTxO entry to push the
  207 -- total above the upper bound.
  208 --
  209 randomImprove
  210     :: (Ord i, Ord o, MonadRandom m)
  211     => CoinSelectionAlgorithm i o m
  212 randomImprove = CoinSelectionAlgorithm payForOutputs
  213 
  214 payForOutputs
  215     :: (Ord i, Ord o, MonadRandom m)
  216     => CoinSelectionParameters i o
  217     -> ExceptT CoinSelectionError m (CoinSelectionResult i o)
  218 payForOutputs params = do
  219     mRandomSelections <- lift $ runMaybeT $ foldM makeRandomSelection
  220         (inputCountMax, inputsAvailable params, []) outputsDescending
  221     case mRandomSelections of
  222         Just (inputCountRemaining, utxoRemaining, randomSelections) -> do
  223             (_, finalSelection, utxoRemaining') <- lift $ foldM
  224                 improveSelection
  225                     (inputCountRemaining, mempty, utxoRemaining)
  226                     (reverse randomSelections)
  227             pure $ CoinSelectionResult finalSelection utxoRemaining'
  228         Nothing ->
  229             throwE errorCondition
  230   where
  231     errorCondition
  232       | amountAvailable < amountRequested =
  233           InputValueInsufficient $
  234               InputValueInsufficientError
  235                   amountAvailable amountRequested
  236       | utxoCount < outputCount =
  237           InputCountInsufficient $
  238               InputCountInsufficientError
  239                   utxoCount outputCount
  240       | utxoCount <= fromIntegral inputCountMax =
  241           InputsExhausted
  242               InputsExhaustedError
  243       | otherwise =
  244           InputLimitExceeded $
  245               InputLimitExceededError $
  246                   fromIntegral inputCountMax
  247     amountAvailable =
  248         coinMapValue $ inputsAvailable params
  249     amountRequested =
  250         coinMapValue $ outputsRequested params
  251     inputCountMax =
  252         fromIntegral $ calculateLimit (limit params) $ fromIntegral outputCount
  253     outputCount =
  254         fromIntegral $ length $ coinMapToList $ outputsRequested params
  255     outputsDescending =
  256         L.sortOn (Down . entryValue) $ coinMapToList $ outputsRequested params
  257     utxoCount =
  258         fromIntegral $ L.length $ coinMapToList $ inputsAvailable params
  259 
  260 -- | Randomly select entries from the given UTxO set, until the total value of
  261 --   selected entries is greater than or equal to the given output value.
  262 --
  263 -- Once a random selection has been made that meets the above criterion, this
  264 -- function returns that selection as is, making no attempt to improve upon
  265 -- the selection in any way.
  266 --
  267 makeRandomSelection
  268     :: forall i o m . MonadRandom m
  269     => (Integer, CoinMap i, [([CoinMapEntry i], CoinMapEntry o)])
  270     -> CoinMapEntry o
  271     -> MaybeT m (Integer, CoinMap i, [([CoinMapEntry i], CoinMapEntry o)])
  272 makeRandomSelection
  273     (inputCountRemaining, utxoRemaining, existingSelections) txout = do
  274         (utxoSelected, utxoRemaining') <- coverRandomly ([], utxoRemaining)
  275         return
  276             ( inputCountRemaining - fromIntegral (L.length utxoSelected)
  277             , utxoRemaining'
  278             , (utxoSelected, txout) : existingSelections
  279             )
  280   where
  281     coverRandomly
  282         :: ([CoinMapEntry i], CoinMap i)
  283         -> MaybeT m ([CoinMapEntry i], CoinMap i)
  284     coverRandomly (selected, remaining)
  285         | L.length selected > fromIntegral inputCountRemaining =
  286             MaybeT $ return Nothing
  287         | sumEntries selected >= targetMin (mkTargetRange txout) =
  288             MaybeT $ return $ Just (selected, remaining)
  289         | otherwise =
  290             utxoPickRandomT remaining >>= \(picked, remaining') ->
  291                 coverRandomly (picked : selected, remaining')

  293 -- | Perform an improvement to random selection on a given output.
  294 improveSelection
  295     :: forall i o m . (MonadRandom m, Ord i, Ord o)
  296     => (Integer, CoinSelection i o, CoinMap i)
  297     -> ([CoinMapEntry i], CoinMapEntry o)
  298     -> m (Integer, CoinSelection i o, CoinMap i)
  299 improveSelection (maxN0, selection, utxo0) (inps0, txout) = do
  300     (maxN, inps, utxo) <- improve (maxN0, inps0, utxo0)
  301     return
  302         ( maxN
  303         , selection <> CoinSelection
  304             { inputs = coinMapFromList inps
  305             , outputs = coinMapFromList [txout]
  306             , change = mkChange txout inps
  307             }
  308         , utxo
  309         )
  310   where
  311     target = mkTargetRange txout
  312 
  313     improve
  314         :: (Integer, [CoinMapEntry i], CoinMap i)
  315         -> m (Integer, [CoinMapEntry i], CoinMap i)
  316     improve (maxN, inps, utxo)
  317         | maxN >= 1 && sumEntries inps < targetAim target = do
  318             runMaybeT (utxoPickRandomT utxo) >>= \case
  319                 Nothing ->
  320                     return (maxN, inps, utxo)
  321                 Just (io, utxo') | isImprovement io inps -> do
  322                     let inps' = io : inps
  323                     let maxN' = maxN - 1
  324                     improve (maxN', inps', utxo')
  325                 Just _ ->
  326                     return (maxN, inps, utxo)
  327         | otherwise =
  328             return (maxN, inps, utxo)
  329 
  330     isImprovement :: CoinMapEntry i -> [CoinMapEntry i] -> Bool
  331     isImprovement io selected =
  332         let
  333             condA = -- (a) It doesn’t exceed a specified upper limit.
  334                 sumEntries (io : selected) < targetMax target
  335 
  336             condB = -- (b) Addition gets us closer to the ideal change
  337                 distanceA < distanceB
  338               where
  339                 distanceA = C.distance
  340                     (targetAim target)
  341                     (sumEntries (io : selected))
  342                 distanceB = C.distance
  343                     (targetAim target)
  344                     (sumEntries selected)
  345 
  346             -- (c) Doesn't exceed maximum number of inputs
  347             -- Guaranteed by the precondition on 'improve'.
  348         in
  349             condA && condB

  355 -- | Represents a target range of /total input values/ for a given output.
  356 --
  357 -- In this context, /total input value/ refers to the total value of a set of
  358 -- inputs selected to pay for a given output.
  359 --
  360 data TargetRange = TargetRange
  361     { targetMin :: Coin
  362         -- ^ The minimum value, corresponding to exactly the requested target
  363         -- amount, and a change amount of zero.
  364     , targetAim :: Coin
  365         -- ^ The ideal value, corresponding to exactly twice the requested
  366         -- target amount, and a change amount equal to the requested amount.
  367     , targetMax :: Coin
  368         -- ^ The maximum value, corresponding to exactly three times the
  369         -- requested amount, and a change amount equal to twice the requested
  370         -- amount.
  371     }
  372 
  373 -- | Compute the target range of /total input values/ for a given output.
  374 --
  375 -- See 'TargetRange'.
  376 --
  377 mkTargetRange :: CoinMapEntry o -> TargetRange
  378 mkTargetRange (CoinMapEntry _ c) = TargetRange
  379     { targetMin = c
  380     , targetAim = c `C.add` c
  381     , targetMax = c `C.add` c `C.add` c
  382     }

  384 -- | Re-wrap 'utxoPickRandom' in a 'MaybeT' monad
  385 utxoPickRandomT
  386     :: MonadRandom m
  387     => CoinMap i
  388     -> MaybeT m (CoinMapEntry i, CoinMap i)
  389 utxoPickRandomT =
  390     MaybeT
  391         . fmap (\(mi, u) -> (, u) <$> mi)
  392         . coinMapRandomEntry

  394 -- | Compute change outputs from a target output and a selection of inputs.
  395 --
  396 -- Pre-condition:
  397 --
  398 -- The output must be less than (or equal to) the sum of the inputs.
  399 --
  400 mkChange :: CoinMapEntry o -> [CoinMapEntry i] -> [Coin]
  401 mkChange (CoinMapEntry _ out) inps =
  402     case difference of
  403         Nothing ->
  404             error $ mconcat
  405                 [ "mkChange: "
  406                 , "output must be less than or equal to sum of inputs"
  407                 ]
  408         Just d | C.isZero d ->
  409             []
  410         Just d ->
  411             [d]
  412   where
  413     difference = sumEntries inps `C.sub` out
