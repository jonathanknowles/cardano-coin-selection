  103 -- | Defines the /maximum/ size of a __dust coin__.
  104 --
  105 -- Functions that accept a 'DustThreshold' argument will generally exclude
  106 -- values that are /less than or equal to/ this threshold from the 'change'
  107 -- sets of generated selections, /coalescing/ such coins together into larger
  108 -- coins that /exceed/ the threshold.
  109 --
  110 -- Specifying a dust threshold of __/n/__ causes all coins that are less than
  111 -- or equal to __/n/__ to be treated as dust and coalesced together.
  112 --
  113 -- Specifying a dust threshold of __0__ completely /disables/ dust elimination
  114 -- with the exception of zero-valued coins, which will always be eliminated.
  115 --
  116 -- See 'coalesceDust'.
  117 --
  118 -- @since 1.0.0
  119 newtype DustThreshold = DustThreshold { unDustThreshold :: Coin }
  120     deriving stock (Eq, Generic, Ord)
  121     deriving Show via (Quiet DustThreshold)
  122 
  123 -- | Returns 'True' if and only if the given 'Coin' is a __dust coin__
  124 --   according to the given 'DustThreshold'.
  125 --
  126 -- A coin is considered to be a dust coin if it is /less than or equal to/
  127 -- the threshold.
  128 --
  129 -- See 'DustThreshold'.
  130 --
  131 isDust :: DustThreshold -> Coin -> Bool
  132 isDust (DustThreshold dt) c = c <= dt

  134 -- | Provides a function capable of __estimating__ the transaction fee required
  135 --   for a given coin selection, according to the rules of a particular
  136 --   blockchain.
  137 --
  138 -- The fee estimate should be a function of the __current__ memberships of the
  139 -- 'inputs', 'outputs', and 'change' sets.
  140 --
  141 -- Depending on the rules of the blockchain under consideration, the fee
  142 -- estimate may take either (or both) of the following factors into account:
  143 --
  144 --   - the number of entries in each set;
  145 --   - the coin value of each set member.
  146 --
  147 -- A fee estimate may differ from the final fee required for a selection, as
  148 -- fees are generally paid for by /adjusting/ a given selection to make a /new/
  149 -- selection. See 'adjustForFee' for more details of this process.
  150 --
  151 -- @since 1.0.0
  152 newtype FeeEstimator i o = FeeEstimator
  153     { estimateFee :: CoinSelection i o -> Fee
  154     } deriving Generic

  179 -- | A choice of fee balancing policies for use when adjusting a coin selection.
  180 --
  181 -- == Background
  182 --
  183 -- A coin selection __'s'__ is said to have a /perfectly-balanced/ fee when it
  184 -- satisfies the following property:
  185 --
  186 -- >>> sumInputs s = sumOutputs s + sumChange s + estimateFee s
  187 --
  188 -- Conversely, a selection is said to have an /unbalanced/ fee when it
  189 -- satisfies the following property:
  190 --
  191 -- >>> sumInputs s > sumOutputs s + sumChange s + estimateFee s
  192 --
  193 -- In other words, if a coin selection has an /unbalanced/ fee, the /effective/
  194 -- fee is greater than the minimum fee /actually required/ by the blockchain.
  195 --
  196 -- == Balanced Fees vs Minimal Fees
  197 --
  198 -- Some blockchains /require /that fees are always /perfectly-balanced/.
  199 --
  200 -- However, for blockchains that allow /unbalanced/ fees, it is sometimes
  201 -- possible to /save money/ by generating a coin selection with an unbalanced
  202 -- fee. This may seem counterintuitive at first, but consider an individual
  203 -- change ouput __/c/__ of value __/v/__. If the /marginal fee/ __/f/__
  204 -- associated with __/c/__ is greater than its value __/v/__, then we will
  205 -- /save money/ by __not__ including __/c/__ within 'change'.
  206 --
  207 -- There are two policy choices available for handling change values with
  208 -- marginal fees greater than their value:
  209 --
  210 --   - For blockchains that __allow__ transactions with /unbalanced/ fees,
  211 --     specifying the 'RequireMinimalFee' policy will allow money to be saved by
  212 --     /excluding/ change outputs that have a marginal fee greater than
  213 --     their value.
  214 --
  215 --   - For blockchains that do __not__ allow transactions with /unbalanced/
  216 --     fees, specifying the 'RequireBalancedFee' policy will always generate
  217 --     selections with fees that are perfectly-balanced, even if the resulting
  218 --     fees are higher than could be achieved by allowing unbalanced fees.
  219 --
  220 data FeeBalancingPolicy
  221     = RequireBalancedFee
  222         -- ^ Generate selections with fees that are perfectly balanced, with the
  223         -- trade-off of allowing slightly higher fees.
  224     | RequireMinimalFee
  225         -- ^ Generate selections with the lowest fees possible, with the
  226         -- trade-off of allowing slightly imbalanced fees.
  227     deriving (Generic, Show, Eq)

  229 -- | Represents the set of possible failures that can occur when adjusting a
  230 --   'CoinSelection' with the 'adjustForFee' function.
  231 --
  232 data FeeAdjustmentError i o
  233     = CannotCoverFee Fee
  234     -- ^ Indicates that the given map of additional inputs was exhausted while
  235     --   attempting to select extra inputs to cover the required fee.
  236     --
  237     -- Records the shortfall (__/f/__ − __/s/__) between the required fee
  238     -- __/f/__ and the total value __/s/__ of currently-selected inputs.
  239 
  240     | CoinSelectionUnderfunded (CoinSelection i o)
  241     -- ^ Indicates that the given coin selection is __underfunded__: the total
  242     -- value of 'inputs' is less than the total value of 'outputs', as
  243     -- calculated by the 'CoinSelection.coinMapValue' function.
  244     deriving (Show, Eq)

  246 -- | Adjusts the given 'CoinSelection' in order to pay for a __transaction__
  247 --   __fee__, required in order to publish the selection as a transaction on
  248 --   a blockchain.
  249 --
  250 -- == Background
  251 --
  252 -- Implementations of 'Cardano.CoinSelection.CoinSelectionAlgorithm' generally
  253 -- produce coin selections that are /exactly balanced/, satisfying the
  254 -- following equality:
  255 --
  256 -- >>> sumInputs s = sumOutputs s + sumChange s
  257 --
  258 -- In order to pay for a transaction fee, the above equality must be
  259 -- transformed into an /inequality/:
  260 --
  261 -- >>> sumInputs s > sumOutputs s + sumChange s
  262 --
  263 -- The difference between these two sides represents value to be paid /by the/
  264 -- /originator/ of the transaction, in the form of a fee:
  265 --
  266 -- >>> sumInputs s = sumOutputs s + sumChange s + fee
  267 --
  268 -- == The Adjustment Process
  269 --
  270 -- In order to generate a fee that is acceptable to the network, this function
  271 -- adjusts the 'change' and 'inputs' of the given 'CoinSelection', consulting
  272 -- the 'FeeEstimator' as a guide for how much the current selection would cost
  273 -- to publish as a transaction on the network.
  274 --
  275 -- == Methods of Adjustment
  276 --
  277 -- There are two methods of adjustment possible:
  278 --
  279 --  1. The __'change'__ set can be /reduced/, either by:
  280 --
  281 --      a. completely removing a change value from the set; or by
  282 --
  283 --      b. reducing a change value to a lower value.
  284 --
  285 --  2. The __'inputs'__ set can be /augmented/, by selecting additional inputs
  286 --     from the specified 'CoinMap' argument.
  287 --
  288 -- == Dealing with Dust Values
  289 --
  290 -- If, at any point, a change value is generated that is less than or equal
  291 -- to the 'DustThreshold', this function will eliminate that change value
  292 -- from the 'change' set, redistributing the eliminated value over the remaining
  293 -- change values, ensuring that the total value of all 'change' is preserved.
  294 --
  295 -- See 'coalesceDust' for more details.
  296 --
  297 -- == Termination
  298 --
  299 -- Since adjusting a selection can affect the fee estimate produced by
  300 -- 'estimateFee', the process of adjustment is an /iterative/ process.
  301 --
  302 -- The termination post-condition depends on the choice of
  303 -- 'FeeBalancingPolicy':
  304 --
  305 --   - If 'RequireBalancedFee' is specified, this function terminates
  306 --     only when it has generated a 'CoinSelection' __'s'__ that satisfies the
  307 --     following property:
  308 --
  309 --         >>> sumInputs s = sumOutputs s + sumChange s + estimateFee s
  310 --
  311 --   - If 'RequireMinimalFee' policy is specified, the above /equality/
  312 --     is relaxed to the following /inequality/:
  313 --
  314 --         >>> sumInputs s ≥ sumOutputs s + sumChange s + estimateFee s
  315 --
  316 -- See 'FeeBalancingPolicy' for more details.
  317 --
  318 -- @since 1.0.0
  319 adjustForFee
  320     :: (Ord i, MonadRandom m)
  321     => FeeOptions i o
  322     -> CoinMap i
  323     -> CoinSelection i o
  324     -> ExceptT (FeeAdjustmentError i o) m (CoinSelection i o)
  325 adjustForFee unsafeOpt utxo coinSel = do
  326     let opt = invariant
  327             "adjustForFee: fee must be non-null" unsafeOpt (not . nullFee)
  328     senderPaysFee opt utxo coinSel
  329   where
  330     nullFee opt = estimateFee (feeEstimator opt) coinSel == Fee C.zero

  336 -- Calculates the current fee associated with a given 'CoinSelection'.
  337 --
  338 -- If the result is less than zero, returns 'Nothing'.
  339 --
  340 calculateFee :: CoinSelection i o -> Maybe Fee
  341 calculateFee s = Fee <$> sumInputs s `C.sub` (sumOutputs s `C.add` sumChange s)
  342 
  343 -- The sender pays fee in this scenario, so fees are removed from the change
  344 -- outputs, and new inputs are selected if necessary.
  345 --
  346 senderPaysFee
  347     :: forall i o m . (Ord i, MonadRandom m)
  348     => FeeOptions i o
  349     -> CoinMap i
  350     -> CoinSelection i o
  351     -> ExceptT (FeeAdjustmentError i o) m (CoinSelection i o)
  352 senderPaysFee opts utxo sel =
  353     evalStateT (go sel) utxo
  354   where
  355     go
  356         :: CoinSelection i o
  357         -> StateT
  358             (CoinMap i)
  359             (ExceptT (FeeAdjustmentError i o) m)
  360             (CoinSelection i o)
  361     go coinSel@(CoinSelection inps outs chgs) = do
  362         -- Substract fee from change outputs, proportionally to their value.
  363         (coinSel', remFee) <- lift $ except $ reduceChangeOutputs opts coinSel
  364 
  365         -- Should the change cover the fee, we're (almost) good. By removing
  366         -- change outputs, we make them smaller and may reduce the size of the
  367         -- transaction, and the fee. Thus, we end up paying slightly more than
  368         -- the upper bound. We could do some binary search and try to
  369         -- re-distribute excess across changes until fee becomes bigger.
  370         if remFee == Fee C.zero
  371         then pure coinSel'
  372         else do
  373             -- Otherwise, we need an extra entries from the available utxo to
  374             -- cover what's left. Note that this entry may increase our change
  375             -- because we may not consume it entirely. So we will just split
  376             -- the extra change across all changes possibly increasing the
  377             -- number of change outputs (if there was none, or if increasing a
  378             -- change value causes an overflow).
  379             --
  380             -- Because selecting a new input increases the fee, we need to
  381             -- re-run the algorithm with this new elements and using the initial
  382             -- change plus the extra change brought up by this entry and see if
  383             -- we can now correctly cover fee.
  384             inps' <- coverRemainingFee remFee
  385             let extraChange = splitCoin (sumEntries inps') chgs
  386             go $ CoinSelection (inps <> coinMapFromList inps') outs extraChange
  387 
  388 -- A short and simple version of the 'random' fee policy to cover for the fee
  389 -- in the case where existing set of change is not enough.
  390 --
  391 coverRemainingFee
  392     :: MonadRandom m
  393     => Fee
  394     -> StateT (CoinMap i) (ExceptT (FeeAdjustmentError i o) m) [CoinMapEntry i]
  395 coverRemainingFee (Fee fee) = go [] where
  396     go acc
  397         | sumEntries acc >= fee =
  398             return acc
  399         | otherwise = do
  400             -- We ignore the size of the fee, and just pick randomly
  401             StateT (lift . coinMapRandomEntry) >>= \case
  402                 Just entry ->
  403                     go (entry : acc)
  404                 Nothing ->
  405                     lift $ throwE $ CannotCoverFee $ Fee $
  406                         fee `C.distance` (sumEntries acc)

  408 -- Pays for the given fee by subtracting it from the given list of change
  409 -- outputs, so that each change output is reduced by a portion of the fee
  410 -- that's in proportion to its relative size.
  411 --
  412 -- == Basic Examples
  413 --
  414 -- >>> reduceChangeOutputs (DustThreshold 0) (Fee 4) (Coin <$> [2, 2, 2, 2])
  415 -- [Coin 1, Coin 1, Coin 1, Coin 1]
  416 --
  417 -- >>> reduceChangeOutputs (DustThreshold 0) (Fee 15) (Coin <$> [2, 4, 8, 16])
  418 -- [Coin 1, Coin 2, Coin 4, Coin 8]
  419 --
  420 -- == Handling Dust
  421 --
  422 -- Any dust outputs in the resulting list are coalesced according to the given
  423 -- dust threshold: (See 'coalesceDust'.)
  424 --
  425 -- >>> reduceChangeOutputs (DustThreshold 1) (Fee 4) (Coin <$> [2, 2, 2, 2])
  426 -- [Coin 4]
  427 --
  428 -- == Handling Insufficient Change
  429 --
  430 -- If there's not enough change to pay for the fee, or if there's only just
  431 -- enough to pay for it exactly, this function returns the /empty list/:
  432 --
  433 -- >>> reduceChangeOutputs (DustThreshold 0) (Fee 15) (Coin <$> [10])
  434 -- []
  435 --
  436 -- >>> reduceChangeOutputs (DustThreshold 0) (Fee 15) (Coin <$> [1, 2, 4, 8])
  437 -- []
  438 --
  439 reduceChangeOutputs
  440     :: FeeOptions i o
  441     -> CoinSelection i o
  442     -> Either (FeeAdjustmentError i o) (CoinSelection i o, Fee)
  443 reduceChangeOutputs opts s = do
  444     -- The original requested fee amount
  445     let Fee φ_original = estimateFee (feeEstimator opts) s
  446     -- The initial amount left for fee (i.e. inputs - outputs)
  447     let mδ_original = sumInputs s `C.sub` (sumOutputs s `C.add` sumChange s)
  448     case mδ_original of
  449         -- selection is now balanced, nothing to do.
  450         Just δ_original | φ_original == δ_original -> do
  451             pure (s, Fee C.zero)
  452 
  453         -- some fee left to pay, but we've depleted all change outputs
  454         Just δ_original | φ_original > δ_original && null (change s) -> do
  455             let remainder = φ_original `C.distance` δ_original
  456             pure (s, Fee remainder)
  457 
  458         -- some fee left to pay, and we've haven't depleted all change yet
  459         Just δ_original | φ_original > δ_original && not (null (change s)) -> do
  460             let remainder = φ_original `C.distance` δ_original
  461             let chgs' = distributeFee (Fee remainder) (NE.fromList (change s))
  462                       & fmap payFee
  463                       & coalesceDust (dustThreshold opts)
  464             reduceChangeOutputs opts (s { change = chgs' })
  465 
  466         -- The current selection has a higher fee than necessary. This typically
  467         -- occurs if, after reducing an output to pay for the predicted fee, the
  468         -- required fee turns out to be less than originally predicted.
  469         -- The outcome depends on whether or not the node allows transactions
  470         -- to be unbalanced.
  471         Just δ_original | δ_original > φ_original -> do
  472             let extraChg = δ_original `C.distance` φ_original
  473             let sDangling = s { change = splitCoin extraChg (change s) }
  474             let Fee φ_dangling = estimateFee (feeEstimator opts) sDangling
  475             -- We have `δ_dangling = φ_original` by construction of sDangling.
  476             --
  477             -- Proof:
  478             --
  479             -- δ_dangling = Σi_dangling - (Σo_dangling + Σc_dangling)
  480             --            = Σi_original - (Σo_original + Σc_original + extraChg)
  481             --            = Σi_original - (Σo_original + Σc_original) - extraChg
  482             --            = δ_original - extraChg
  483             --            = δ_original - (δ_original - φ_original)
  484             --            = φ_original
  485             let δ_dangling = φ_original
  486             case φ_dangling `C.sub` δ_dangling of
  487                 -- we've left too much, but adding a change output would be more
  488                 -- expensive than not having it. Here we have two choices:
  489                 --
  490                 -- a) If the node allows unbalanced transaction, we can stop
  491                 --    here and do nothing. We'll leave slightly more than what's
  492                 --    needed for fees, but having an extra change output isn't
  493                 --    worth it anyway.
  494                 --
  495                 -- b) If we __must__ balance the transaction, then we can choose
  496                 --    to pay the extra cost by adding the change output and
  497                 --    continue trying to balance the transaction (likely, by
  498                 --    selecting another input).
  499                 Just remainder | φ_dangling >= δ_original ->
  500                     case feeBalancingPolicy opts of
  501                         RequireMinimalFee ->
  502                             pure (s, Fee C.zero)
  503                         RequireBalancedFee ->
  504                             pure (sDangling, Fee remainder)
  505 
  506                 -- If however, adding the dangling change doesn't cost more than
  507                 -- not having it, we might as well add it to get the money and
  508                 -- continue balancing!
  509                 _otherwise ->
  510                     reduceChangeOutputs opts sDangling
  511 
  512         -- The only way to end-up here is if the user has provided an invalid
  513         -- selection where outputs are trying to spend more than inputs. This is
  514         -- simply forbidden.
  515         _Nothing ->
  516             Left (CoinSelectionUnderfunded s)
  517 
  518 -- Distribute the given fee over the given list of coins, so that each coin
  519 -- is allocated a __fraction__ of the fee in proportion to its relative size.
  520 --
  521 -- == Pre-condition
  522 --
  523 -- Every coin in the given list must be __non-zero__ in value.
  524 --
  525 -- == Examples
  526 --
  527 -- >>> distributeFee (Fee 2) [(Coin 1), (Coin 1)]
  528 -- [(Fee 1, Coin 1), (Fee 1, Coin 1)]
  529 --
  530 -- >>> distributeFee (Fee 4) [(Coin 1), (Coin 1)]
  531 -- [(Fee 2, Coin 1), (Fee 2, Coin 1)]
  532 --
  533 -- >>> distributeFee (Fee 7) [(Coin 1), (Coin 2), (Coin 4)]
  534 -- [(Fee 1, Coin 1), (Fee 2, Coin 2), (Fee 4, Coin 4)]
  535 --
  536 -- >>> distributeFee (Fee 14) [(Coin 1), (Coin 2), (Coin 4)]
  537 -- [(Fee 2, Coin 1), (Fee 4, Coin 2), (Fee 8, Coin 4)]
  538 --
  539 distributeFee :: Fee -> NonEmpty Coin -> NonEmpty (Fee, Coin)
  540 distributeFee (Fee feeTotal) coinsUnsafe =
  541     NE.zip feesRounded coins
  542   where
  543     -- A list of coins that are non-zero in value.
  544     coins :: NonEmpty Coin
  545     coins =
  546         invariant "distributeFee: all coins must be non-zero in value."
  547         coinsUnsafe (C.zero `F.notElem`)
  548 
  549     -- A list of rounded fee portions, where each fee portion deviates from the
  550     -- ideal unrounded portion by as small an amount as possible.
  551     feesRounded :: NonEmpty Fee
  552     feesRounded
  553         -- 1. Start with the list of ideal unrounded fee portions for each coin:
  554         = feesUnrounded
  555         -- 2. Attach an index to each fee portion, so that we can remember the
  556         --    original order:
  557         & NE.zip indices
  558         -- 3. Sort the fees into descending order of their fractional parts:
  559         & NE.sortBy (comparing (Down . fractionalPart . snd))
  560         -- 4. Apply pre-computed roundings to each fee portion:
  561         --    * portions with the greatest fractional parts are rounded up;
  562         --    * portions with the smallest fractional parts are rounded down.
  563         & NE.zipWith (\roundDir (i, f) -> (i, round roundDir f)) feeRoundings
  564         -- 5. Restore the original order:
  565         & NE.sortBy (comparing fst)
  566         -- 6. Strip away the indices:
  567         & fmap snd
  568         -- 7. Transform results into fees:
  569         & fmap (Fee . fromMaybe C.zero . C.coinFromIntegral @Integer)
  570       where
  571         indices :: NonEmpty Int
  572         indices = 0 :| [1 ..]
  573 
  574     -- A list of rounding directions, one per fee portion.
  575     --
  576     -- Since the ideal fee portion for each coin is a rational value, we must
  577     -- therefore round each rational value either /up/ or /down/ to produce a
  578     -- final integer result.
  579     --
  580     -- However, we can't take the simple approach of either rounding /all/ fee
  581     -- portions down or rounding /all/ fee portions up, as this could cause the
  582     -- sum of fee portions to either undershoot or overshoot the original fee.
  583     --
  584     -- So in order to hit the fee exactly, we must round /some/ of the portions
  585     -- up, and /some/ of the portions down.
  586     --
  587     -- Fortunately, we can calculate exactly how many fee portions must be
  588     -- rounded up, by first rounding /all/ portions down, and then computing
  589     -- the /shortfall/ between the sum of the rounded-down portions and the
  590     -- original fee.
  591     --
  592     -- We return a list where all values of 'RoundUp' occur in a contiguous
  593     -- section at the start of the list, of the following form:
  594     --
  595     --     [RoundUp, RoundUp, ..., RoundDown, RoundDown, ...]
  596     --
  597     feeRoundings :: NonEmpty RoundingDirection
  598     feeRoundings =
  599         applyN feeShortfall (NE.cons RoundUp) (NE.repeat RoundDown)
  600       where
  601          -- The part of the total fee that we'd lose if we were to take the
  602          -- simple approach of rounding all ideal fee portions /down/.
  603         feeShortfall
  604             = C.coinToIntegral feeTotal
  605             - fromIntegral @Integer (F.sum $ round RoundDown <$> feesUnrounded)
  606 
  607     -- A list of ideal unrounded fee portions, with one fee portion per coin.
  608     --
  609     -- A coin's ideal fee portion is the rational portion of the total fee that
  610     -- corresponds to that coin's relative size when compared to other coins.
  611     feesUnrounded :: NonEmpty Rational
  612     feesUnrounded = calculateIdealFee <$> coins
  613       where
  614         calculateIdealFee c
  615             = C.coinToIntegral c
  616             * C.coinToIntegral feeTotal
  617             % C.coinToIntegral totalCoinValue
  618 
  619     -- The total value of all coins.
  620     totalCoinValue :: Coin
  621     totalCoinValue = F.fold coins
  622 
  623 -- | From the given list of coins, remove dust coins with a value less than or
  624 --   equal to the given threshold value, redistributing their total value over
  625 --   the coins that remain.
  626 --
  627 -- This function satisfies the following properties:
  628 --
  629 -- >>> sum coins = sum (coalesceDust threshold coins)
  630 -- >>> all (/= Coin 0) (coalesceDust threshold coins)
  631 --
  632 coalesceDust :: DustThreshold -> NonEmpty Coin -> [Coin]
  633 coalesceDust threshold coins =
  634     splitCoin valueToDistribute coinsToKeep
  635   where
  636     (coinsToRemove, coinsToKeep) = NE.partition (isDust threshold) coins
  637     valueToDistribute = F.fold coinsToRemove
  638 
  639 -- Splits up the given coin of value __@v@__, distributing its value over the
  640 -- given coin list of length __@n@__, so that each coin value is increased by
  641 -- an integral amount within unity of __@v/n@__, producing a new list of coin
  642 -- values where the overall total is preserved.
  643 --
  644 -- == Basic Examples
  645 --
  646 -- When it's possible to divide a coin evenly, each coin value is increased by
  647 -- the same integer amount:
  648 --
  649 -- >>> splitCoin (Coin 40) (Coin <$> [1, 1, 1, 1])
  650 -- [Coin 11, Coin 11, Coin 11, Coin 11]
  651 --
  652 -- >>> splitCoin (Coin 40) (Coin <$> [1, 2, 3, 4])
  653 -- [Coin 11, Coin 12, Coin 13, Coin 14]
  654 --
  655 -- == Handling Non-Uniform Increases
  656 --
  657 -- When it's not possible to divide a coin evenly, each integral coin value in
  658 -- the resulting list is always within unity of the ideal unrounded result:
  659 --
  660 -- >>> splitCoin (Coin 2) (Coin <$> [1, 1, 1, 1])
  661 -- [Coin 1, Coin 1, Coin 2, Coin 2]
  662 --
  663 -- >>> splitCoin (Coin 10) (Coin <$> [1, 1, 1, 1])
  664 -- [Coin 3, Coin 3, Coin 4, Coin 4]
  665 --
  666 -- == Handling Empty Lists
  667 --
  668 -- If the given list is empty, this function returns a list with the original
  669 -- given coin as its sole element:
  670 --
  671 -- >>> splitCoin (Coin 10) []
  672 -- [Coin 10]
  673 --
  674 -- == Properties
  675 --
  676 -- The total value is always preserved:
  677 --
  678 -- >>> sum (splitCoin x ys) == x + sum ys
  679 --
  680 splitCoin :: Coin -> [Coin] -> [Coin]
  681 splitCoin coinToSplit coinsToIncrease =
  682     case (mIncrement, mShortfall) of
  683         (Just increment, Just shortfall) ->
  684             zipWith C.add coinsToIncrease increments
  685           where
  686             increments = zipWith C.add majorIncrements minorIncrements
  687             majorIncrements = repeat increment
  688             minorIncrements = replicate (C.coinToIntegral shortfall) C.one
  689                 <> repeat C.zero
  690         _ | coinToSplit > C.zero ->
  691             [coinToSplit]
  692         _ ->
  693             []
  694   where
  695     mCoinCount = length coinsToIncrease
  696     mIncrement = coinToSplit `C.div` mCoinCount
  697     mShortfall = coinToSplit `C.mod` mCoinCount
  698 
  699 -- Extract the fractional part of a rational number.
  700 --
  701 -- Examples:
  702 --
  703 -- >>> fractionalPart (3 % 2)
  704 -- 1 % 2
  705 --
  706 -- >>> fractionalPart (11 % 10)
  707 -- 1 % 10
  708 --
  709 fractionalPart :: Rational -> Rational
  710 fractionalPart = snd . properFraction @_ @Integer
