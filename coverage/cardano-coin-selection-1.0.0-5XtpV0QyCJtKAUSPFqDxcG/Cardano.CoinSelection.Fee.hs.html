  102 -- | Defines the /maximum/ size of a __dust coin__.
  103 --
  104 -- Functions that accept a 'DustThreshold' argument will generally exclude
  105 -- values that are /less than or equal to/ this threshold from the 'change'
  106 -- sets of generated selections, /coalescing/ such coins together into larger
  107 -- coins that /exceed/ the threshold.
  108 --
  109 -- Specifying a dust threshold of __/n/__ causes all coins that are less than
  110 -- or equal to __/n/__ to be treated as dust and coalesced together.
  111 --
  112 -- Specifying a dust threshold of __0__ completely /disables/ dust elimination
  113 -- with the exception of zero-valued coins, which will always be eliminated.
  114 --
  115 -- See 'coalesceDust'.
  116 --
  117 -- @since 1.0.0
  118 newtype DustThreshold = DustThreshold { unDustThreshold :: Coin }
  119     deriving stock (Eq, Generic, Ord)
  120     deriving Show via (Quiet DustThreshold)

  122 -- | Provides a function capable of __estimating__ the transaction fee required
  123 --   for a given coin selection, according to the rules of a particular
  124 --   blockchain.
  125 --
  126 -- The fee estimate should be a function of the __current__ memberships of the
  127 -- 'inputs', 'outputs', and 'change' sets.
  128 --
  129 -- Depending on the rules of the blockchain under consideration, the fee
  130 -- estimate may take either (or both) of the following factors into account:
  131 --
  132 --   - the number of entries in each set;
  133 --   - the coin value of each set member.
  134 --
  135 -- A fee estimate may differ from the final fee required for a selection, as
  136 -- fees are generally paid for by /adjusting/ a given selection to make a /new/
  137 -- selection. See 'adjustForFee' for more details of this process.
  138 --
  139 -- @since 1.0.0
  140 newtype FeeEstimator i o = FeeEstimator
  141     { estimateFee :: CoinSelection i o -> Fee
  142     } deriving Generic

  167 -- | A choice of fee balancing policies for use when adjusting a coin selection.
  168 --
  169 -- == Background
  170 --
  171 -- A coin selection __'s'__ is said to have a /perfectly-balanced/ fee when it
  172 -- satisfies the following property:
  173 --
  174 -- >>> sumInputs s = sumOutputs s + sumChange s + estimateFee s
  175 --
  176 -- Conversely, a selection is said to have an /unbalanced/ fee when it
  177 -- satisfies the following property:
  178 --
  179 -- >>> sumInputs s > sumOutputs s + sumChange s + estimateFee s
  180 --
  181 -- In other words, if a coin selection has an /unbalanced/ fee, the /effective/
  182 -- fee is greater than the minimum fee /actually required/ by the blockchain.
  183 --
  184 -- == Balanced Fees vs Minimal Fees
  185 --
  186 -- Some blockchains /require /that fees are always /perfectly-balanced/.
  187 --
  188 -- However, for blockchains that allow /unbalanced/ fees, it is sometimes
  189 -- possible to /save money/ by generating a coin selection with an unbalanced
  190 -- fee. This may seem counterintuitive at first, but consider an individual
  191 -- change ouput __/c/__ of value __/v/__. If the /marginal fee/ __/f/__
  192 -- associated with __/c/__ is greater than its value __/v/__, then we will
  193 -- /save money/ by __not__ including __/c/__ within 'change'.
  194 --
  195 -- There are two policy choices available for handling change values with
  196 -- marginal fees greater than their value:
  197 --
  198 --   - For blockchains that __allow__ transactions with /unbalanced/ fees,
  199 --     specifying the 'RequireMinimalFee' policy will allow money to be saved by
  200 --     /excluding/ change outputs that have a marginal fee greater than
  201 --     their value.
  202 --
  203 --   - For blockchains that do __not__ allow transactions with /unbalanced/
  204 --     fees, specifying the 'RequireBalancedFee' policy will always generate
  205 --     selections with fees that are perfectly-balanced, even if the resulting
  206 --     fees are higher than could be achieved by allowing unbalanced fees.
  207 --
  208 data FeeBalancingPolicy
  209     = RequireBalancedFee
  210         -- ^ Generate selections with fees that are perfectly balanced, with the
  211         -- trade-off of allowing slightly higher fees.
  212     | RequireMinimalFee
  213         -- ^ Generate selections with the lowest fees possible, with the
  214         -- trade-off of allowing slightly imbalanced fees.
  215     deriving (Generic, Show, Eq)

  217 -- | Represents the set of possible failures that can occur when adjusting a
  218 --   'CoinSelection' with the 'adjustForFee' function.
  219 --
  220 data FeeAdjustmentError i o
  221     = CannotCoverFee Fee
  222     -- ^ Indicates that the given map of additional inputs was exhausted while
  223     --   attempting to select extra inputs to cover the required fee.
  224     --
  225     -- Records the shortfall (__/f/__ − __/s/__) between the required fee
  226     -- __/f/__ and the total value __/s/__ of currently-selected inputs.
  227 
  228     | CoinSelectionUnderfunded (CoinSelection i o)
  229     -- ^ Indicates that the given coin selection is __underfunded__: the total
  230     -- value of 'inputs' is less than the total value of 'outputs', as
  231     -- calculated by the 'CoinSelection.coinMapValue' function.
  232     deriving (Show, Eq)

  234 -- | Adjusts the given 'CoinSelection' in order to pay for a __transaction__
  235 --   __fee__, required in order to publish the selection as a transaction on
  236 --   a blockchain.
  237 --
  238 -- == Background
  239 --
  240 -- Implementations of 'Cardano.CoinSelection.CoinSelectionAlgorithm' generally
  241 -- produce coin selections that are /exactly balanced/, satisfying the
  242 -- following equality:
  243 --
  244 -- >>> sumInputs s = sumOutputs s + sumChange s
  245 --
  246 -- In order to pay for a transaction fee, the above equality must be
  247 -- transformed into an /inequality/:
  248 --
  249 -- >>> sumInputs s > sumOutputs s + sumChange s
  250 --
  251 -- The difference between these two sides represents value to be paid /by the/
  252 -- /originator/ of the transaction, in the form of a fee:
  253 --
  254 -- >>> sumInputs s = sumOutputs s + sumChange s + fee
  255 --
  256 -- == The Adjustment Process
  257 --
  258 -- In order to generate a fee that is acceptable to the network, this function
  259 -- adjusts the 'change' and 'inputs' of the given 'CoinSelection', consulting
  260 -- the 'FeeEstimator' as a guide for how much the current selection would cost
  261 -- to publish as a transaction on the network.
  262 --
  263 -- == Methods of Adjustment
  264 --
  265 -- There are two methods of adjustment possible:
  266 --
  267 --  1. The __'change'__ set can be /reduced/, either by:
  268 --
  269 --      a. completely removing a change value from the set; or by
  270 --
  271 --      b. reducing a change value to a lower value.
  272 --
  273 --  2. The __'inputs'__ set can be /augmented/, by selecting additional inputs
  274 --     from the specified 'CoinMap' argument.
  275 --
  276 -- == Dealing with Dust Values
  277 --
  278 -- If, at any point, a change value is generated that is less than or equal
  279 -- to the 'DustThreshold', this function will eliminate that change value
  280 -- from the 'change' set, redistributing the eliminated value over the remaining
  281 -- change values, ensuring that the total value of all 'change' is preserved.
  282 --
  283 -- See 'coalesceDust' for more details.
  284 --
  285 -- == Termination
  286 --
  287 -- Since adjusting a selection can affect the fee estimate produced by
  288 -- 'estimateFee', the process of adjustment is an /iterative/ process.
  289 --
  290 -- The termination post-condition depends on the choice of
  291 -- 'FeeBalancingPolicy':
  292 --
  293 --   - If 'RequireBalancedFee' is specified, this function terminates
  294 --     only when it has generated a 'CoinSelection' __'s'__ that satisfies the
  295 --     following property:
  296 --
  297 --         >>> sumInputs s = sumOutputs s + sumChange s + estimateFee s
  298 --
  299 --   - If 'RequireMinimalFee' policy is specified, the above /equality/
  300 --     is relaxed to the following /inequality/:
  301 --
  302 --         >>> sumInputs s ≥ sumOutputs s + sumChange s + estimateFee s
  303 --
  304 -- See 'FeeBalancingPolicy' for more details.
  305 --
  306 -- @since 1.0.0
  307 adjustForFee
  308     :: (Ord i, MonadRandom m)
  309     => FeeOptions i o
  310     -> CoinMap i
  311     -> CoinSelection i o
  312     -> ExceptT (FeeAdjustmentError i o) m (CoinSelection i o)
  313 adjustForFee unsafeOpt utxo coinSel = do
  314     let opt = invariant
  315             "adjustForFee: fee must be non-null" unsafeOpt (not . nullFee)
  316     senderPaysFee opt utxo coinSel
  317   where
  318     nullFee opt = estimateFee (feeEstimator opt) coinSel == Fee C.zero

  324 -- Calculates the current fee associated with a given 'CoinSelection'.
  325 --
  326 -- If the result is less than zero, returns 'Nothing'.
  327 --
  328 calculateFee :: CoinSelection i o -> Maybe Fee
  329 calculateFee s = Fee <$> sumInputs s `C.sub` (sumOutputs s `C.add` sumChange s)
  330 
  331 -- The sender pays fee in this scenario, so fees are removed from the change
  332 -- outputs, and new inputs are selected if necessary.
  333 --
  334 senderPaysFee
  335     :: forall i o m . (Ord i, MonadRandom m)
  336     => FeeOptions i o
  337     -> CoinMap i
  338     -> CoinSelection i o
  339     -> ExceptT (FeeAdjustmentError i o) m (CoinSelection i o)
  340 senderPaysFee opts utxo sel =
  341     evalStateT (go sel) utxo
  342   where
  343     go
  344         :: CoinSelection i o
  345         -> StateT
  346             (CoinMap i)
  347             (ExceptT (FeeAdjustmentError i o) m)
  348             (CoinSelection i o)
  349     go coinSel@(CoinSelection inps outs chgs) = do
  350         -- Substract fee from change outputs, proportionally to their value.
  351         (coinSel', remFee) <- lift $ except $ reduceChangeOutputs opts coinSel
  352 
  353         -- Should the change cover the fee, we're (almost) good. By removing
  354         -- change outputs, we make them smaller and may reduce the size of the
  355         -- transaction, and the fee. Thus, we end up paying slightly more than
  356         -- the upper bound. We could do some binary search and try to
  357         -- re-distribute excess across changes until fee becomes bigger.
  358         if remFee == Fee C.zero
  359         then pure coinSel'
  360         else do
  361             -- Otherwise, we need an extra entries from the available utxo to
  362             -- cover what's left. Note that this entry may increase our change
  363             -- because we may not consume it entirely. So we will just split
  364             -- the extra change across all changes possibly increasing the
  365             -- number of change outputs (if there was none, or if increasing a
  366             -- change value causes an overflow).
  367             --
  368             -- Because selecting a new input increases the fee, we need to
  369             -- re-run the algorithm with this new elements and using the initial
  370             -- change plus the extra change brought up by this entry and see if
  371             -- we can now correctly cover fee.
  372             inps' <- coverRemainingFee remFee
  373             let extraChange = splitCoin (sumEntries inps') chgs
  374             go $ CoinSelection (inps <> coinMapFromList inps') outs extraChange
  375 
  376 -- A short and simple version of the 'random' fee policy to cover for the fee
  377 -- in the case where existing set of change is not enough.
  378 --
  379 coverRemainingFee
  380     :: MonadRandom m
  381     => Fee
  382     -> StateT (CoinMap i) (ExceptT (FeeAdjustmentError i o) m) [CoinMapEntry i]
  383 coverRemainingFee (Fee fee) = go [] where
  384     go acc
  385         | sumEntries acc >= fee =
  386             return acc
  387         | otherwise = do
  388             -- We ignore the size of the fee, and just pick randomly
  389             StateT (lift . coinMapRandomEntry) >>= \case
  390                 Just entry ->
  391                     go (entry : acc)
  392                 Nothing ->
  393                     lift $ throwE $ CannotCoverFee $ Fee $
  394                         fee `C.distance` (sumEntries acc)

  396 -- Pays for the given fee by subtracting it from the given list of change
  397 -- outputs, so that each change output is reduced by a portion of the fee
  398 -- that's in proportion to its relative size.
  399 --
  400 -- == Basic Examples
  401 --
  402 -- >>> reduceChangeOutputs (DustThreshold 0) (Fee 4) (Coin <$> [2, 2, 2, 2])
  403 -- [Coin 1, Coin 1, Coin 1, Coin 1]
  404 --
  405 -- >>> reduceChangeOutputs (DustThreshold 0) (Fee 15) (Coin <$> [2, 4, 8, 16])
  406 -- [Coin 1, Coin 2, Coin 4, Coin 8]
  407 --
  408 -- == Handling Dust
  409 --
  410 -- Any dust outputs in the resulting list are coalesced according to the given
  411 -- dust threshold: (See 'coalesceDust'.)
  412 --
  413 -- >>> reduceChangeOutputs (DustThreshold 1) (Fee 4) (Coin <$> [2, 2, 2, 2])
  414 -- [Coin 4]
  415 --
  416 -- == Handling Insufficient Change
  417 --
  418 -- If there's not enough change to pay for the fee, or if there's only just
  419 -- enough to pay for it exactly, this function returns the /empty list/:
  420 --
  421 -- >>> reduceChangeOutputs (DustThreshold 0) (Fee 15) (Coin <$> [10])
  422 -- []
  423 --
  424 -- >>> reduceChangeOutputs (DustThreshold 0) (Fee 15) (Coin <$> [1, 2, 4, 8])
  425 -- []
  426 --
  427 reduceChangeOutputs
  428     :: FeeOptions i o
  429     -> CoinSelection i o
  430     -> Either (FeeAdjustmentError i o) (CoinSelection i o, Fee)
  431 reduceChangeOutputs opts s = do
  432     -- The original requested fee amount
  433     let Fee φ_original = estimateFee (feeEstimator opts) s
  434     -- The initial amount left for fee (i.e. inputs - outputs)
  435     let mδ_original = sumInputs s `C.sub` (sumOutputs s `C.add` sumChange s)
  436     case mδ_original of
  437         -- selection is now balanced, nothing to do.
  438         Just δ_original | φ_original == δ_original -> do
  439             pure (s, Fee C.zero)
  440 
  441         -- some fee left to pay, but we've depleted all change outputs
  442         Just δ_original | φ_original > δ_original && null (change s) -> do
  443             let remainder = φ_original `C.distance` δ_original
  444             pure (s, Fee remainder)
  445 
  446         -- some fee left to pay, and we've haven't depleted all change yet
  447         Just δ_original | φ_original > δ_original && not (null (change s)) -> do
  448             let remainder = φ_original `C.distance` δ_original
  449             let chgs' = distributeFee (Fee remainder) (NE.fromList (change s))
  450                       & fmap payFee
  451                       & coalesceDust (dustThreshold opts)
  452             reduceChangeOutputs opts (s { change = chgs' })
  453 
  454         -- The current selection has a higher fee than necessary. This typically
  455         -- occurs if, after reducing an output to pay for the predicted fee, the
  456         -- required fee turns out to be less than originally predicted.
  457         -- The outcome depends on whether or not the node allows transactions
  458         -- to be unbalanced.
  459         Just δ_original | δ_original > φ_original -> do
  460             let extraChg = δ_original `C.distance` φ_original
  461             let sDangling = s { change = splitCoin extraChg (change s) }
  462             let Fee φ_dangling = estimateFee (feeEstimator opts) sDangling
  463             -- We have `δ_dangling = φ_original` by construction of sDangling.
  464             --
  465             -- Proof:
  466             --
  467             -- δ_dangling = Σi_dangling - (Σo_dangling + Σc_dangling)
  468             --            = Σi_original - (Σo_original + Σc_original + extraChg)
  469             --            = Σi_original - (Σo_original + Σc_original) - extraChg
  470             --            = δ_original - extraChg
  471             --            = δ_original - (δ_original - φ_original)
  472             --            = φ_original
  473             let δ_dangling = φ_original
  474             case φ_dangling `C.sub` δ_dangling of
  475                 -- we've left too much, but adding a change output would be more
  476                 -- expensive than not having it. Here we have two choices:
  477                 --
  478                 -- a) If the node allows unbalanced transaction, we can stop
  479                 --    here and do nothing. We'll leave slightly more than what's
  480                 --    needed for fees, but having an extra change output isn't
  481                 --    worth it anyway.
  482                 --
  483                 -- b) If we __must__ balance the transaction, then we can choose
  484                 --    to pay the extra cost by adding the change output and
  485                 --    continue trying to balance the transaction (likely, by
  486                 --    selecting another input).
  487                 Just remainder | φ_dangling >= δ_original ->
  488                     case feeBalancingPolicy opts of
  489                         RequireMinimalFee ->
  490                             pure (s, Fee C.zero)
  491                         RequireBalancedFee ->
  492                             pure (sDangling, Fee remainder)
  493 
  494                 -- If however, adding the dangling change doesn't cost more than
  495                 -- not having it, we might as well add it to get the money and
  496                 -- continue balancing!
  497                 _otherwise ->
  498                     reduceChangeOutputs opts sDangling
  499 
  500         -- The only way to end-up here is if the user has provided an invalid
  501         -- selection where outputs are trying to spend more than inputs. This is
  502         -- simply forbidden.
  503         _Nothing ->
  504             Left (CoinSelectionUnderfunded s)
  505 
  506 -- Distribute the given fee over the given list of coins, so that each coin
  507 -- is allocated a __fraction__ of the fee in proportion to its relative size.
  508 --
  509 -- == Pre-condition
  510 --
  511 -- Every coin in the given list must be __non-zero__ in value.
  512 --
  513 -- == Examples
  514 --
  515 -- >>> distributeFee (Fee 2) [(Coin 1), (Coin 1)]
  516 -- [(Fee 1, Coin 1), (Fee 1, Coin 1)]
  517 --
  518 -- >>> distributeFee (Fee 4) [(Coin 1), (Coin 1)]
  519 -- [(Fee 2, Coin 1), (Fee 2, Coin 1)]
  520 --
  521 -- >>> distributeFee (Fee 7) [(Coin 1), (Coin 2), (Coin 4)]
  522 -- [(Fee 1, Coin 1), (Fee 2, Coin 2), (Fee 4, Coin 4)]
  523 --
  524 -- >>> distributeFee (Fee 14) [(Coin 1), (Coin 2), (Coin 4)]
  525 -- [(Fee 2, Coin 1), (Fee 4, Coin 2), (Fee 8, Coin 4)]
  526 --
  527 distributeFee :: Fee -> NonEmpty Coin -> NonEmpty (Fee, Coin)
  528 distributeFee (Fee feeTotal) coinsUnsafe =
  529     NE.zip feesRounded coins
  530   where
  531     -- A list of coins that are non-zero in value.
  532     coins :: NonEmpty Coin
  533     coins =
  534         invariant "distributeFee: all coins must be non-zero in value."
  535         coinsUnsafe (C.zero `F.notElem`)
  536 
  537     -- A list of rounded fee portions, where each fee portion deviates from the
  538     -- ideal unrounded portion by as small an amount as possible.
  539     feesRounded :: NonEmpty Fee
  540     feesRounded
  541         -- 1. Start with the list of ideal unrounded fee portions for each coin:
  542         = feesUnrounded
  543         -- 2. Attach an index to each fee portion, so that we can remember the
  544         --    original order:
  545         & NE.zip indices
  546         -- 3. Sort the fees into descending order of their fractional parts:
  547         & NE.sortBy (comparing (Down . fractionalPart . snd))
  548         -- 4. Apply pre-computed roundings to each fee portion:
  549         --    * portions with the greatest fractional parts are rounded up;
  550         --    * portions with the smallest fractional parts are rounded down.
  551         & NE.zipWith (\roundDir (i, f) -> (i, round roundDir f)) feeRoundings
  552         -- 5. Restore the original order:
  553         & NE.sortBy (comparing fst)
  554         -- 6. Strip away the indices:
  555         & fmap snd
  556         -- 7. Transform results into fees:
  557         & fmap (Fee . fromMaybe C.zero . C.coinFromIntegral @Integer)
  558       where
  559         indices :: NonEmpty Int
  560         indices = 0 :| [1 ..]
  561 
  562     -- A list of rounding directions, one per fee portion.
  563     --
  564     -- Since the ideal fee portion for each coin is a rational value, we must
  565     -- therefore round each rational value either /up/ or /down/ to produce a
  566     -- final integer result.
  567     --
  568     -- However, we can't take the simple approach of either rounding /all/ fee
  569     -- portions down or rounding /all/ fee portions up, as this could cause the
  570     -- sum of fee portions to either undershoot or overshoot the original fee.
  571     --
  572     -- So in order to hit the fee exactly, we must round /some/ of the portions
  573     -- up, and /some/ of the portions down.
  574     --
  575     -- Fortunately, we can calculate exactly how many fee portions must be
  576     -- rounded up, by first rounding /all/ portions down, and then computing
  577     -- the /shortfall/ between the sum of the rounded-down portions and the
  578     -- original fee.
  579     --
  580     -- We return a list where all values of 'RoundUp' occur in a contiguous
  581     -- section at the start of the list, of the following form:
  582     --
  583     --     [RoundUp, RoundUp, ..., RoundDown, RoundDown, ...]
  584     --
  585     feeRoundings :: NonEmpty RoundingDirection
  586     feeRoundings =
  587         applyN feeShortfall (NE.cons RoundUp) (NE.repeat RoundDown)
  588       where
  589          -- The part of the total fee that we'd lose if we were to take the
  590          -- simple approach of rounding all ideal fee portions /down/.
  591         feeShortfall
  592             = C.coinToIntegral feeTotal
  593             - fromIntegral @Integer (F.sum $ round RoundDown <$> feesUnrounded)
  594 
  595     -- A list of ideal unrounded fee portions, with one fee portion per coin.
  596     --
  597     -- A coin's ideal fee portion is the rational portion of the total fee that
  598     -- corresponds to that coin's relative size when compared to other coins.
  599     feesUnrounded :: NonEmpty Rational
  600     feesUnrounded = calculateIdealFee <$> coins
  601       where
  602         calculateIdealFee c
  603             = C.coinToIntegral c
  604             * C.coinToIntegral feeTotal
  605             % C.coinToIntegral totalCoinValue
  606 
  607     -- The total value of all coins.
  608     totalCoinValue :: Coin
  609     totalCoinValue = F.fold coins

  611 -- | From the given list of coins, remove dust coins with a value less than or
  612 --   equal to the given threshold value, redistributing their total value over
  613 --   the coins that remain.
  614 --
  615 -- This function satisfies the following properties:
  616 --
  617 -- >>> sum coins = sum (coalesceDust threshold coins)
  618 -- >>> all (/= Coin 0) (coalesceDust threshold coins)
  619 --
  620 coalesceDust :: DustThreshold -> NonEmpty Coin -> [Coin]
  621 coalesceDust (DustThreshold threshold) coins =
  622     splitCoin valueToDistribute coinsToKeep
  623   where
  624     (coinsToKeep, coinsToRemove) = NE.partition (> threshold) coins
  625     valueToDistribute = F.fold coinsToRemove
  626 
  627 -- Splits up the given coin of value __@v@__, distributing its value over the
  628 -- given coin list of length __@n@__, so that each coin value is increased by
  629 -- an integral amount within unity of __@v/n@__, producing a new list of coin
  630 -- values where the overall total is preserved.
  631 --
  632 -- == Basic Examples
  633 --
  634 -- When it's possible to divide a coin evenly, each coin value is increased by
  635 -- the same integer amount:
  636 --
  637 -- >>> splitCoin (Coin 40) (Coin <$> [1, 1, 1, 1])
  638 -- [Coin 11, Coin 11, Coin 11, Coin 11]
  639 --
  640 -- >>> splitCoin (Coin 40) (Coin <$> [1, 2, 3, 4])
  641 -- [Coin 11, Coin 12, Coin 13, Coin 14]
  642 --
  643 -- == Handling Non-Uniform Increases
  644 --
  645 -- When it's not possible to divide a coin evenly, each integral coin value in
  646 -- the resulting list is always within unity of the ideal unrounded result:
  647 --
  648 -- >>> splitCoin (Coin 2) (Coin <$> [1, 1, 1, 1])
  649 -- [Coin 1, Coin 1, Coin 2, Coin 2]
  650 --
  651 -- >>> splitCoin (Coin 10) (Coin <$> [1, 1, 1, 1])
  652 -- [Coin 3, Coin 3, Coin 4, Coin 4]
  653 --
  654 -- == Handling Empty Lists
  655 --
  656 -- If the given list is empty, this function returns a list with the original
  657 -- given coin as its sole element:
  658 --
  659 -- >>> splitCoin (Coin 10) []
  660 -- [Coin 10]
  661 --
  662 -- == Properties
  663 --
  664 -- The total value is always preserved:
  665 --
  666 -- >>> sum (splitCoin x ys) == x + sum ys
  667 --
  668 splitCoin :: Coin -> [Coin] -> [Coin]
  669 splitCoin coinToSplit coinsToIncrease =
  670     case (mIncrement, mShortfall) of
  671         (Just increment, Just shortfall) ->
  672             zipWith C.add coinsToIncrease increments
  673           where
  674             increments = zipWith C.add majorIncrements minorIncrements
  675             majorIncrements = repeat increment
  676             minorIncrements = replicate (C.coinToIntegral shortfall) C.one
  677                 <> repeat C.zero
  678         _ | coinToSplit > C.zero ->
  679             [coinToSplit]
  680         _ ->
  681             []
  682   where
  683     mCoinCount = length coinsToIncrease
  684     mIncrement = coinToSplit `C.div` mCoinCount
  685     mShortfall = coinToSplit `C.mod` mCoinCount
  686

  689 -- Examples:

  691 -- >>> fractionalPart (3 % 2)
  692 -- 1 % 2
  693 --
  694 -- >>> fractionalPart (11 % 10)

  710 -- | Reduce a coin value by a given fee amount. If fees are too big for
  711 -- a single coin, returns a `Coin 0`.
